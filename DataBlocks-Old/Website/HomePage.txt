                "We build Digital Transformation_"
            ];
            
            let phraseIndex = 0;
            let charIndex = 0;
            let isDeleting = false;
            let typingSpeed = 100; // Base typing speed in ms
            
            function typeWriter() {
                const currentPhrase = phrases[phraseIndex];
                
                if (isDeleting) {
                    // Deleting characters
                    typewriterElement.textContent = currentPhrase.substring(0, charIndex - 1);
                    charIndex--;
                    typingSpeed = 50; // Faster when deleting
                } else {
                    // Typing characters
                    typewriterElement.textContent = currentPhrase.substring(0, charIndex + 1);
                    charIndex++;
                    typingSpeed = 100; // Normal speed when typing
                }
                
                // If we've completed typing the current phrase
                if (!isDeleting && charIndex === currentPhrase.length) {
                    // Pause at the end of typing
                    isDeleting = true;
                    typingSpeed = 2000; // Wait a bit before deleting
                } 
                // If we've deleted the entire phrase
                else if (isDeleting && charIndex === 0) {
                    isDeleting = false;
                    phraseIndex++;
                    if (phraseIndex === phrases.length) {
                        phraseIndex = 0;
                    }
                    typingSpeed = 100; // Wait between phrases
                }
                
                setTimeout(typeWriter, typingSpeed);
            }
            
            typeWriter();
        });
    </script>
